A = [1, 2, 3;
     1/2, 1, 4;
     1/3, 1/4, 1];

% 计算每个准则的权重
n = size(A, 1);
w = zeros(n, 1);
for i = 1:n
    w(i) = prod(A(i,:))^(1/n);
end
w = w / sum(w);

% 计算一致性比例
CI = (max(eig(A)) - n) / (n - 1);
RI = [0, 0, 0.58, 0.90, 1.12, 1.24, 1.32, 1.41, 1.45];
CR = CI / RI(n);

% 输出结果
fprintf('权重向量 w:\n');
disp(w');
fprintf('一致性指标 CI: %.4f\n', CI);
fprintf('一致性比例 CR: %.4f\n', CR);
